परिचय

MiniExcel .NET के लिए एक सरल और कुशल Excel प्रोसेसिंग टूल है, जिसे विशेष रूप से न्यूनतम मेमोरी उपयोग के लिए डिज़ाइन किया गया है।

वर्तमान में, अधिकांश लोकप्रिय फ्रेमवर्क्स को ऑपरेशन को आसान बनाने के लिए Excel दस्तावेज़ से सभी डेटा को मेमोरी में लोड करना पड़ता है, लेकिन इससे मेमोरी की खपत की समस्या हो सकती है। MiniExcel का तरीका अलग है: डेटा को स्ट्रीमिंग तरीके से पंक्ति दर पंक्ति प्रोसेस किया जाता है, जिससे मूल खपत को सैकड़ों मेगाबाइट्स से घटाकर केवल कुछ मेगाबाइट्स कर दिया जाता है, जिससे आउट-ऑफ-मेमोरी (OOM) समस्याओं को प्रभावी ढंग से रोका जा सकता है।

विशेषताएँ

  • मेमोरी खपत को न्यूनतम करता है, आउट-ऑफ-मेमोरी (OOM) त्रुटियों को रोकता है और पूर्ण गार्बेज कलेक्शन्स से बचाता है
  • बड़े डेटासेट्स पर बेहतर प्रदर्शन के लिए वास्तविक समय, पंक्ति-स्तरीय डेटा ऑपरेशंस सक्षम करता है
  • डिफर्ड एक्जीक्यूशन के साथ LINQ को सपोर्ट करता है, जिससे तेज़, मेमोरी-कुशल पेजिंग और जटिल क्वेरीज़ संभव हैं
  • हल्का, जिसमें Microsoft Office या COM+ घटकों की आवश्यकता नहीं है, और DLL आकार 500KB से कम है
  • एक्सेल को पढ़ने/लिखने/भरने के लिए सरल और सहज API शैली

12. डिस्क-बेस कैश द्वारा बड़े फ़ाइल पढ़ना (Disk-Base Cache - SharedString)

यदि SharedStrings का आकार 5 MB से अधिक हो जाता है, तो MiniExcel डिफ़ॉल्ट रूप से स्थानीय डिस्क कैश का उपयोग करेगा, उदाहरण के लिए, 10x100000.xlsx (दस लाख पंक्तियों का डेटा), जब डिस्क कैश अक्षम हो तो अधिकतम मेमोरी उपयोग 195MB है, लेकिन सक्षम डिस्क कैश के साथ केवल 65MB की आवश्यकता होती है। ध्यान दें, इस ऑप्टिमाइज़ेशन में कुछ दक्षता लागत होती है, इसलिए इस स्थिति में पढ़ने का समय 7.4 सेकंड से बढ़कर 27.2 सेकंड हो जाता है। यदि आपको इसकी आवश्यकता नहीं है तो आप निम्नलिखित कोड द्वारा डिस्क कैश अक्षम कर सकते हैं:

var config = new OpenXmlConfiguration { EnableSharedStringCache = false };
MiniExcel.Query(path,configuration: config)

आप SharedStringCacheSize का उपयोग करके साझा किए गए स्ट्रिंग फ़ाइल आकार को निर्दिष्ट आकार के आगे बढ़ने पर डिस्क कैशिंग के लिए बदल सकते हैं

var config = new OpenXmlConfiguration { SharedStringCacheSize=500*1024*1024 };
MiniExcel.Query(path, configuration: config);

private IEnumerable<Dictionary<string, object>> GetOrders(string lang, string role, Order[] orders)
{
    foreach (var order in orders)
    {
        var newOrder = new Dictionary<string, object>();

        if (lang == "zh-CN")
        {
            newOrder.Add("客户编号", order.CustomerID);
            newOrder.Add("订单编号", order.OrderNo);
            newOrder.Add("产品编号", order.ProductID);
            newOrder.Add("数量", order.Qty);
            if (role == "Sales")
                newOrder.Add("价格", order.Amt);

    }

private IEnumerable<Dictionary<int, object>> ConvertToIntIndexRows(IEnumerable<object> rows)
{
    ICollection<string> keys = null;
    var isFirst = true;
    foreach (IDictionary<string,object> r in rows)
    {
        if(isFirst)
        {
            keys = r.Keys;
            isFirst = false;
        }

        var dic = new Dictionary<int, object>();
        var index = 0;
        foreach (var key in keys)
            dic[index++] = r[key];
        yield return dic;
    }
}

public static DataTable QueryAsDataTableWithoutEmptyRow(Stream stream, bool useHeaderRow, string sheetName, ExcelType excelType, string startCell, IConfiguration configuration)
{
    if (sheetName == null && excelType != ExcelType.CSV) /*Issue #279*/
        sheetName = stream.GetSheetNames().First();

    var dt = new DataTable(sheetName);
    var first = true;
    var rows = stream.Query(useHeaderRow,sheetName,excelType,startCell,configuration);
    foreach (IDictionary<string, object> row in rows)
    {
        if (first)
        {

            foreach (var key in row.Keys)
            {
                var column = new DataColumn(key, typeof(object)) { Caption = key };

                dt.Columns.Add(column);
            }

            dt.BeginLoadData();
            first = false;
        }

        var newRow = dt.NewRow();
        var isNull=true;
        foreach (var key in row.Keys)
        {
            var _v = row[key];
            if(_v!=null)
                isNull = false;
            newRow[key] = _v;
        }

        if(!isNull)
            dt.Rows.Add(newRow);
    }

    dt.EndLoadData();
    return dt;
}
